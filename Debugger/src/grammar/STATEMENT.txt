STATEMENT -> if EXPRESSION IF_STATEMENT STATEMENT |
             while EXPRESSION WHILE_STATEMENT STATEMENT |
             SIMPLE_STATEMENT STATEMENT |
             #

IF_STATEMENT -> {STATEMENT} [else ELSE_STATEMENT]
ELSE_STATEMENT -> {STATEMENT}
WHILE_STATEMENT -> {STATEMENT}
SIMPLE_STATEMENT -> EXPRESSION;
EXPRESSION -> + | - | ++ |
